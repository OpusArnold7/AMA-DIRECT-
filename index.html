CREATE TABLE IF NOT EXISTS captain_applications (id bigint generated always as identity primary key, name text, phone text, email text, state text, role text, why text, captain_id text, status text default 'active', created_at timestamptz default now()); CREATE TABLE IF NOT EXISTS wholesaler_applications (id bigint generated always as identity primary key, name text, phone text, state text, product text, capacity text, ws_id text, status text default 'pending', created_at timestamptz default now());
